# VocalCoach requirements — CUDA GPU build
# Installs CPU requirements first, then upgrades torch to the CUDA build.
#
# Usage (choose one matching your CUDA version):
#
#   CUDA 12.1  (RTX 30xx / 40xx, most modern GPUs):
#     pip install -r requirements-gpu.txt
#     pip install torch==2.3.0+cu121 torchaudio==2.3.0+cu121 \
#         --index-url https://download.pytorch.org/whl/cu121
#
#   CUDA 11.8  (older drivers):
#     pip install torch==2.3.0+cu118 torchaudio==2.3.0+cu118 \
#         --index-url https://download.pytorch.org/whl/cu118
#
#   Check your CUDA version: nvcc --version

# Include all CPU requirements
-r requirements.txt

# ── CUDA extras ─────────────────────────────────────────────────────────────
# (torch + torchaudio are overridden by the pip install command above)
# No additional packages required — CUDA support is bundled in the torch wheel.
